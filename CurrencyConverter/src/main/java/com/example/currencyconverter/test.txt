package com.example.currencyconverter;

import javafx.application.Application;
import javafx.fxml.FXMLLoader;
import javafx.geometry.Insets;
import javafx.scene.Node;
import javafx.scene.Scene;
import javafx.scene.control.*;
import javafx.scene.layout.VBox;
import javafx.stage.Stage;
import javafx.util.converter.DoubleStringConverter;

import java.io.IOException;
import java.util.ArrayList;
import java.util.Collection;

public class CurrencyConverterApplication extends Application {

    Stage window;
    Scene scene;
    private String currTo;
    private String currFrom;
    private double inputAmount;

    @Override
    public void start(Stage stage) throws IOException {
        window = stage;
        stage.setTitle("Currency Converter");

        ArrayList<Node> elements = new ArrayList<>();

        Label title = new Label("Welcome to Currency Converter V1.0");
        Label desc = new Label("Please choose your currencies below:");

        ChoiceBox<String> currencyFrom = new ChoiceBox<>();
        ChoiceBox<String> currencyTo = new ChoiceBox<>();
        Label fromLabel = new Label("Converting from:" , currencyFrom);
        Label toLabel = new Label("Converting to:" , currencyTo);

        TextField amountInput = new TextField();
        amountInput.setTextFormatter(new TextFormatter<>(new DoubleStringConverter()));

        Button calculate = new Button("Go!");

        elements.add(title);
        elements.add(desc);
        elements.add(currencyFrom);
        elements.add(currencyTo);
        elements.add(fromLabel);
        elements.add(toLabel);
        elements.add(amountInput);
        elements.add(calculate);

        VBox layout = new VBox(30);
        layout.setPadding(new Insets(20,20,20,20));
        layout.getChildren().addAll(title,desc,currencyFrom,currencyTo,fromLabel,toLabel,amountInput,calculate);

        stage.setScene(scene);
        stage.show();
    }

    public static void main(String[] args) {
        launch();
    }
